add_custom_command(
    OUTPUT ${CMAKE_BINARY_DIR}/contrib/libhubbub/src/treebuilder/element-type.gperf
    COMMAND ${CMAKE_COMMAND} -E make_directory ${CMAKE_BINARY_DIR}/contrib/libhubbub/src/treebuilder
    COMMAND ${PYTHON3} ${CMAKE_SOURCE_DIR}/utils/cmake_helper.py convert-eol "${CMAKE_SOURCE_DIR}/contrib/libhubbub/src/treebuilder/element-type.gperf" "${CMAKE_BINARY_DIR}/contrib/libhubbub/src/treebuilder/element-type.gperf"
    DEPENDS ${CMAKE_SOURCE_DIR}/contrib/libhubbub/src/treebuilder/element-type.gperf
)
add_custom_command(
    OUTPUT ${CMAKE_BINARY_DIR}/contrib/libhubbub/src/treebuilder/autogenerated-element-type.c
    COMMAND ${GPERF} ${CMAKE_BINARY_DIR}/contrib/libhubbub/src/treebuilder/element-type.gperf --output-file=${CMAKE_BINARY_DIR}/contrib/libhubbub/src/treebuilder/autogenerated-element-type.c
    DEPENDS ${CMAKE_BINARY_DIR}/contrib/libhubbub/src/treebuilder/element-type.gperf
)
